<script>
let saved = JSON.parse(localStorage.getItem("osrsTracker")) || {};
let stepsData = [];

async function loadJSON() {
  try {
    const res = await fetch("steps.json");
    stepsData = await res.json();
    render();
  } catch {
    console.warn("No steps.json found.");
  }
}

function render() {
  const app = document.getElementById("app");
  app.innerHTML = "";

  let total = 0, completed = 0;

  stepsData.forEach((sec) => {
    const sectionDiv = document.createElement("div");
    sectionDiv.className = "section";

    const title = document.createElement("h2");
    title.textContent = sec.section;
    sectionDiv.appendChild(title);

    sec.items.forEach((item) => {
      const key = item;
      const isChecked = saved[key] || false;

      total++;
      if (isChecked) completed++;

      const div = document.createElement("div");
      div.className = "step" + (isChecked ? " completed" : "");

      const checkbox = document.createElement("input");
      checkbox.type = "checkbox";
      checkbox.checked = isChecked;

      checkbox.onchange = () => {
        saved[key] = checkbox.checked;
        localStorage.setItem("osrsTracker", JSON.stringify(saved));
        render();
      };

      const label = document.createElement("span");
      label.textContent = item;

      div.appendChild(checkbox);
      div.appendChild(label);
      sectionDiv.appendChild(div);
    });

    app.appendChild(sectionDiv);
  });

  document.getElementById("progressBar").style.width =
    (total ? (completed / total) * 100 : 0) + "%";
}

// Convert wiki-style text into JSON
function convertWikiText() {
  const raw = document.getElementById("wikiInput").value;
  const lines = raw.split("\n");

  let result = [];
  let currentSection = null;

  lines.forEach(line => {
    line = line.trim();
    if (!line) return;

    // Section detection
    if (line.startsWith("==") || line.endsWith(":")) {
      const clean = line.replace(/=+/g, "").replace(":", "").trim();
      currentSection = { section: clean, items: [] };
      result.push(currentSection);
      return;
    }

    // Bullet points
    if (line.startsWith("*") || line.startsWith("-") || line.startsWith("•")) {
      const clean = line.replace(/^[*\-•]\s*/, "");
      if (currentSection) currentSection.items.push(clean);
      return;
    }

    // Fallback
    if (currentSection) {
      currentSection.items.push(line);
    }
  });

  document.getElementById("jsonInput").value =
    JSON.stringify(result, null, 2);
}

function loadFromInput() {
  try {
    stepsData = JSON.parse(document.getElementById("jsonInput").value);
    render();
  } catch {
    alert("Invalid JSON format");
  }
}

function resetProgress() {
  localStorage.removeItem("osrsTracker");
  saved = {};
  render();
}

loadJSON();
</script>
